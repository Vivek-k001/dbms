DELIMITER //

CREATE PROCEDURE membership_status(
    IN cid INT,
    OUT membership VARCHAR(20)
)
BEGIN
    DECLARE cred INT;

    SELECT credits INTO cred
    FROM customer
    WHERE customer_id = cid;

    IF cred > 5000 THEN
        SET membership = 'PLATINUM';
    ELSEIF cred >= 1000 AND cred <= 5000 THEN
        SET membership = 'GOLD';
    ELSE
        SET membership = 'SILVER';
    END IF;

END //

DELIMITER ;
